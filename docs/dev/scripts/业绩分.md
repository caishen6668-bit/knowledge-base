# 业绩分.py

## 功能说明

计算员工业绩得分。读取业绩 Excel 文件，根据各阶段（D0/D1/S1）回款达成率，按区间规则换算成分数。

## 完整代码

```python
import pandas as pd
import os

# === 1. 路径和文件名 ===
desktop = os.path.join(os.path.expanduser("~"), "Desktop")
perf_file = os.path.join(desktop, "17-23数.xlsx")      # 回收业绩表（已改名）
out_file = os.path.join(desktop, "业绩达成率.xlsx")     # 输出文件名

# === 2. 读取业绩表 ===
df = pd.read_excel(perf_file)

# 如需检查列名，可以临时打开注释：
# print(df.columns)

# === 3. 字段统一命名 ===
df = df.rename(columns={
    '催员姓名': 'agent',
    'statis_date(day)': 'date',
    '逾期阶段': 'stage',
    '达成率': 'ach_rate',
    '订单催回率': 'order_rate'   # 用于筛选订单催回率≥100%的记录
})

# === 3.1 清洗达成率，转成可计算的数字 ===
# 先转成字符串，去空格，去掉可能存在的百分号 %
s_ach = df['ach_rate'].astype(str).str.strip().str.replace('%', '', regex=False)
df['ach_rate'] = pd.to_numeric(s_ach, errors='coerce')

# === 3.2 清洗订单催回率，转成可计算的数字，用于剔除 >=100% 的记录 ===
# 同样先去掉百分号等
s_order = df['order_rate'].astype(str).str.strip().str.replace('%', '', regex=False)
df['order_rate'] = pd.to_numeric(s_order, errors='coerce')

# === 3.3 剔除：订单催回率 >= 100% 的记录 ===
# 情况1：表里是小数形式（1.0 = 100%），则条件用 >= 1
# 情况2：表里是百分数字符（"100%"），上面转成 100，再根据需要 /100 处理
# 这里按“小数形式”为主，直接用 < 1 保留 <100% 的记录
df = df[df['order_rate'] < 1]

# 去掉没有姓名或没有达成率的行
df = df.dropna(subset=['agent', 'ach_rate'])

# === 4. 计算每个催员的综合达成率（用达成率求平均） ===
perf = df.groupby('agent')['ach_rate'].mean().reset_index()
perf = perf.rename(columns={'ach_rate': '综合达成率'})

# === 5. 按综合达成率从高到低排序并生成名次 ===
perf = perf.sort_values('综合达成率', ascending=False).reset_index(drop=True)
perf['rank'] = perf.index + 1  # 名次从 1 开始

N = len(perf)

# === 6. 计算每个档位的人数（四舍五入） ===
# 第1档：前20% → 60分
# 第2档：21–40% → 50分
# 第3档：41–60% → 40分
# 第4档：61–80% → 30分
# 第5档：81–100% → 20分

n1 = round(N * 0.2)
n2 = round(N * 0.2)
n3 = round(N * 0.2)
n4 = round(N * 0.2)
n5 = max(0, N - (n1 + n2 + n3 + n4))  # 剩余给第5档（n5只是校验用，不直接用）

def score_by_rank(r):
    if r <= n1:
        return 60
    elif r <= n1 + n2:
        return 50
    elif r <= n1 + n2 + n3:
        return 40
    elif r <= n1 + n2 + n3 + n4:
        return 30
    else:
        return 20

perf['业绩分'] = perf['rank'].apply(score_by_rank)

# === 7. 导出时只保留百分比形式的综合达成率 ===
perf['综合达成率_百分比'] = (perf['综合达成率'] * 100).round(2).astype(str) + '%'

# 只导出百分比，不导出小数形式
result = perf[['agent', '综合达成率_百分比', 'rank', '业绩分']]

# === 8. 保存结果到桌面 ===
result.to_excel(out_file, index=False)

print("✅ 已剔除订单催回率≥100%的记录，业绩达成率与业绩分计算完成，保存到：", out_file)
print(result.head())
```
