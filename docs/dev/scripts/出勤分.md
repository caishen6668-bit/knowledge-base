# 出勤分.py

## 功能说明

计算员工出勤得分。读取出勤 Excel，根据缺勤天数按规则换算出勤分数。

## 完整代码

```python
import pandas as pd
import os

# === 1. 路径和文件名 ===
desktop = os.path.join(os.path.expanduser("~"), "Desktop")
att_file = os.path.join(desktop, "20-26出勤.xlsx")  # <- 出勤表文件名，你可以改

# === 2. 读取出勤表 ===
att = pd.read_excel(att_file)

# === 3. 重命名字段，方便后续统一处理 ===
att = att.rename(columns={
    '员工姓名': 'agent',
    '排班日期(day)': 'date',
    '排班状态': 'status'
})

# === 4. 根据排班状态，标记是否缺勤 ===
# 状态只有：上班 / 休息 / 请假 / 缺勤
att['is_absent'] = att['status'].apply(lambda x: 1 if x == '缺勤' else 0)

# === 5. 按人统计缺勤次数 ===
absent_count = att.groupby('agent')['is_absent'].sum().reset_index()
absent_count = absent_count.rename(columns={'is_absent': '缺勤次数'})

# === 6. 按规则换算出勤分 ===
def calc_att_score(n_absent):
    if n_absent == 0:
        return 20
    elif n_absent == 1:
        return 10
    else:
        return 0

absent_count['出勤分'] = absent_count['缺勤次数'].apply(calc_att_score)

# === 7. 排个序看看（可选） ===
result_att = absent_count.sort_values('出勤分', ascending=False)

# === 8. 保存到桌面（可改文件名）===
out_file = os.path.join(desktop, "出勤得分.xlsx")
result_att.to_excel(out_file, index=False)

print("✅ 出勤分已计算完成，结果文件：出勤得分.xlsx 已保存在桌面")
print(result_att.head())
```
