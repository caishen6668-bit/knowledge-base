    周度日度生成.py

    功能说明

    生成周度和日度报表。读取原始数据，按日和周维度汇总，生成格式化报表。

    完整代码

    import openpyxl
from openpyxl import Workbook
from openpyxl.utils import get_column_letter
from openpyxl.styles import Font, Alignment, PatternFill, Border, Side

ratios = [
    ("D-2", "新客", 0.11),
    ("D-2", "老客", 0.10),
    ("D-1", "新客", 0.12),
    ("D-1", "老客", 0.16),
    ("D0",  "新客", 0.16),
    ("D0",  "老客", 0.22),
    ("S1",  "新客", 0.22),
    ("S1",  "老客", 0.25),
]
param_lookup_range = "参数!$D$2:$E$9"

detail_rows = [
    ("D-2", "YNT301", "新客"),
    ("D-2", "YNT301", "老客"),
    ("D-2", "YNT302", "新客"),
    ("D-2", "YNT302", "老客"),
    ("D-1", "YNT101", "新客"),
    ("D-1", "YNT101", "老客"),
    ("D0", "YND001", "新客"),
    ("D0", "YND001", "老客"),
    ("S1", "YNS101", "新客"),
    ("S1", "YNS101", "老客"),
]
stage_list = ["D-2", "D-1", "D0", "S1"]
dates = ["2025/12/13", "2025/12/12", "2025/12/11", "2025/12/10", "2025/12/09", "2025/12/08"]
sub_headers_cn = ["案件数", "回收数", "目标", "回收率"]
base_headers_cn = ["阶段", "账号", "客群", "周度目标", "周度回收", "周度达成率"]
sub_headers_en = ["Case Count", "Recovery Count", "Target", "Recovery Rate"]
base_headers_en = ["Stage", "Account", "Customer Type", "Weekly Target", "Weekly Recovery", "Weekly Achieving Rate"]

wb = Workbook()

# ========= Sheet1：参数 ==========
param = wb.active
param.title = "参数"
param_headers = ["阶段", "客群", "比例", "键", "比例2"]
param.append(param_headers)
for stage, cust, rate in ratios:
    row = param.max_row + 1
    param[f"A{row}"] = stage
    param[f"B{row}"] = cust
    param[f"C{row}"] = rate
    param[f"D{row}"] = stage + cust
    param[f"E{row}"] = rate

# ========= Sheet2：中文周度合计 ==========
ws_cn = wb.create_sheet("中文周度合计")
for col, h in enumerate(base_headers_cn, start=1):
    ws_cn.cell(row=1, column=col, value=h)
    ws_cn.cell(row=2, column=col, value=None)
for di, date in enumerate(dates):
    start_col = 7 + di * 4
    ws_cn.merge_cells(start_row=1, start_column=start_col, end_row=1, end_column=start_col + 3)
    ws_cn.cell(row=1, column=start_col, value=date)
    for offset, sub in enumerate(sub_headers_cn):
        ws_cn.cell(row=2, column=start_col + offset, value=sub)
for col in range(1, 7):
    ws_cn.merge_cells(start_row=1, start_column=col, end_row=2, end_column=col)

case_cols = []
recovery_cols = []
target_cols = []
rate_cols = []
for di in range(len(dates)):
    sc = 7 + di * 4
    case_cols.append(get_column_letter(sc))
    recovery_cols.append(get_column_letter(sc+1))
    target_cols.append(get_column_letter(sc+2))
    rate_cols.append(get_column_letter(sc+3))

start_row = 3
row_cn = start_row
summary_rows_cn = []

for stage in stage_list:
    # 获取当前阶段所有账号
    stage_accounts = [r[1] for r in detail_rows if r[0] == stage]
    stage_accounts = sorted(list(set(stage_accounts)))
    for account in stage_accounts:
        # 新客
        ws_cn[f"A{row_cn}"] = stage
        ws_cn[f"B{row_cn}"] = account
        ws_cn[f"C{row_cn}"] = "新客"
        # 老客
        ws_cn[f"A{row_cn+1}"] = stage
        ws_cn[f"B{row_cn+1}"] = account
        ws_cn[f"C{row_cn+1}"] = "老客"
        # 汇总
        ws_cn[f"A{row_cn+2}"] = stage
        ws_cn[f"B{row_cn+2}"] = account
        ws_cn[f"C{row_cn+2}"] = "汇总"
        summary_rows_cn.append(row_cn+2)

        # 日度公式
        for di in range(len(dates)):
            case_col = case_cols[di]
            recovery_col = recovery_cols[di]
            target_col = target_cols[di]
            rate_col = rate_cols[di]
            # 新客目标
            ws_cn[f"{target_col}{row_cn}"] = (
                f'=IF({case_col}{row_cn}=0,0,'
                f'IF($C{row_cn}="汇总","",'
                f'MAX(1,ROUND({case_col}{row_cn}*VLOOKUP($A{row_cn}&$C{row_cn},{param_lookup_range},2,FALSE),0))))'
            )
            # 老客目标
            ws_cn[f"{target_col}{row_cn+1}"] = (
                f'=IF({case_col}{row_cn+1}=0,0,'
                f'IF($C{row_cn+1}="汇总","",'
                f'MAX(1,ROUND({case_col}{row_cn+1}*VLOOKUP($A{row_cn+1}&$C{row_cn+1},{param_lookup_range},2,FALSE),0))))'
            )
            # 汇总公式
            ws_cn[f"{case_col}{row_cn+2}"] = f"={case_col}{row_cn}+{case_col}{row_cn+1}"
            ws_cn[f"{recovery_col}{row_cn+2}"] = f"={recovery_col}{row_cn}+{recovery_col}{row_cn+1}"
            ws_cn[f"{target_col}{row_cn+2}"] = f"={target_col}{row_cn}+{target_col}{row_cn+1}"
            ws_cn[f"{rate_col}{row_cn+2}"] = f'=IF({target_col}{row_cn+2}=0,"",{recovery_col}{row_cn+2}/{target_col}{row_cn+2})'
            # 回收率
            ws_cn[f"{rate_col}{row_cn}"] = f'=IF({target_col}{row_cn}=0,"",{recovery_col}{row_cn}/{target_col}{row_cn})'
            ws_cn[f"{rate_col}{row_cn+1}"] = f'=IF({target_col}{row_cn+1}=0,"",{recovery_col}{row_cn+1}/{target_col}{row_cn+1})'

        # 周度公式
        sum_cases_new = ",".join(f"{c}{row_cn}" for c in case_cols)
        sum_cases_old = ",".join(f"{c}{row_cn+1}" for c in case_cols)
        ws_cn[f"D{row_cn}"] = (
            f'=IF($C{row_cn}="汇总","",'
            f'IF(SUM({sum_cases_new})=0,0,'
            f'ROUND(SUM({sum_cases_new})*VLOOKUP($A{row_cn}&$C{row_cn},{param_lookup_range},2,FALSE),0)))'
        )
        ws_cn[f"D{row_cn+1}"] = (
            f'=IF($C{row_cn+1}="汇总","",'
            f'IF(SUM({sum_cases_old})=0,0,'
            f'ROUND(SUM({sum_cases_old})*VLOOKUP($A{row_cn+1}&$C{row_cn+1},{param_lookup_range},2,FALSE),0)))'
        )
        ws_cn[f"D{row_cn+2}"] = f"=D{row_cn}+D{row_cn+1}"
        # 周度回收
        sum_recs_new = ",".join(f"{c}{row_cn}" for c in recovery_cols)
        sum_recs_old = ",".join(f"{c}{row_cn+1}" for c in recovery_cols)
        ws_cn[f"E{row_cn}"] = f"=SUM({sum_recs_new})"
        ws_cn[f"E{row_cn+1}"] = f"=SUM({sum_recs_old})"
        ws_cn[f"E{row_cn+2}"] = f"=E{row_cn}+E{row_cn+1}"
        # 周度达成率
        ws_cn[f"F{row_cn}"] = f'=IF(D{row_cn}=0,"",E{row_cn}/D{row_cn})'
        ws_cn[f"F{row_cn+1}"] = f'=IF(D{row_cn+1}=0,"",E{row_cn+1}/D{row_cn+1})'
        ws_cn[f"F{row_cn+2}"] = f'=IF(D{row_cn+2}=0,"",E{row_cn+2}/D{row_cn+2})'

        row_cn += 3

# ========= English Weekly Summary ==========
ws_en = wb.create_sheet("English Weekly Summary")
for col, h in enumerate(base_headers_en, start=1):
    ws_en.cell(row=1, column=col, value=h)

    ws_en.cell(row=1, column=start_col, value=date)
    for offset, sub in enumerate(sub_headers_en):
        ws_en.cell(row=2, column=start_col + offset, value=sub)
for col in range(1, 7):
    ws_en.merge_cells(start_row=1, start_column=col, end_row=2, end_column=col)

case_cols_en = []
recovery_cols_en = []
target_cols_en = []
rate_cols_en = []
for di in range(len(dates)):
    sc = 7 + di * 4
    case_cols_en.append(get_column_letter(sc))
    recovery_cols_en.append(get_column_letter(sc+1))
    target_cols_en.append(get_column_letter(sc+2))
    rate_cols_en.append(get_column_letter(sc+3))

row_en = start_row
summary_rows_en = []
for stage in stage_list:
    stage_accounts = [r[1] for r in detail_rows if r[0] == stage]
    stage_accounts = sorted(list(set(stage_accounts)))
    for account in stage_accounts:
        # 新客
        ws_en[f"A{row_en}"] = stage
        ws_en[f"B{row_en}"] = account
        ws_en[f"C{row_en}"] = "New"
        # 老客
        ws_en[f"A{row_en+1}"] = stage
        ws_en[f"B{row_en+1}"] = account
        ws_en[f"C{row_en+1}"] = "Old"
        # 汇总
        ws_en[f"A{row_en+2}"] = stage
        ws_en[f"B{row_en+2}"] = account
        ws_en[f"C{row_en+2}"] = "Summary"
        summary_rows_en.append(row_en+2)
        for di in range(len(dates)):
            case_col = case_cols_en[di]
            recovery_col = recovery_cols_en[di]
            target_col = target_cols_en[di]
            rate_col = rate_cols_en[di]
            ws_en[f"{case_col}{row_en}"] = f"='中文周度合计'!{case_col}{row_en}"
            ws_en[f"{case_col}{row_en+1}"] = f"='中文周度合计'!{case_col}{row_en+1}"
            ws_en[f"{case_col}{row_en+2}"] = f"='中文周度合计'!{case_col}{row_en+2}"
            ws_en[f"{recovery_col}{row_en}"] = f"='中文周度合计'!{recovery_col}{row_en}"
            ws_en[f"{recovery_col}{row_en+1}"] = f"='中文周度合计'!{recovery_col}{row_en+1}"
            ws_en[f"{recovery_col}{row_en+2}"] = f"='中文周度合计'!{recovery_col}{row_en+2}"
            ws_en[f"{target_col}{row_en}"] = f"='中文周度合计'!{target_col}{row_en}"
            ws_en[f"{target_col}{row_en+1}"] = f"='中文周度合计'!{target_col}{row_en+1}"
            ws_en[f"{target_col}{row_en+2}"] = f"='中文周度合计'!{target_col}{row_en+2}"
            ws_en[f"{rate_col}{row_en}"] = f"='中文周度合计'!{rate_col}{row_en}"
            ws_en[f"{rate_col}{row_en+1}"] = f"='中文周度合计'!{rate_col}{row_en+1}"
            ws_en[f"{rate_col}{row_en+2}"] = f"='中文周度合计'!{rate_col}{row_en+2}"
        ws_en[f"D{row_en}"] = f"='中文周度合计'!D{row_en}"
        ws_en[f"D{row_en+1}"] = f"='中文周度合计'!D{row_en+1}"
        ws_en[f"D{row_en+2}"] = f"='中文周度合计'!D{row_en+2}"
        ws_en[f"E{row_en}"] = f"='中文周度合计'!E{row_en}"
        ws_en[f"E{row_en+1}"] = f"='中文周度合计'!E{row_en+1}"
        ws_en[f"E{row_en+2}"] = f"='中文周度合计'!E{row_en+2}"
        ws_en[f"F{row_en}"] = f"='中文周度合计'!F{row_en}"
        ws_en[f"F{row_en+1}"] = f"='中文周度合计'!F{row_en+1}"
        ws_en[f"F{row_en+2}"] = f"='中文周度合计'!F{row_en+2}"
        row_en += 3

def beautify(ws, summary_rows, rate_cols):
    bold_font = Font(bold=True)
    center_align = Alignment(horizontal="center", vertical="center")
    fill_summary = PatternFill("solid", fgColor="FFD966")
    fill_header = PatternFill("solid", fgColor="BDD7EE")
    fill_weekly = PatternFill("solid", fgColor="C6EFCE")
    thin = Side(border_style="thin", color="000000")
    border = Border(left=thin, right=thin, top=thin, bottom=thin)

    for r in range(1, ws.max_row + 1):
        for c in range(1, ws.max_column + 1):
            cell = ws.cell(row=r, column=c)
            cell.alignment = center_align
            cell.border = border

    for col in range(1, ws.max_column + 1):
        ws.cell(row=1, column=col).font = bold_font
        ws.cell(row=1, column=col).fill = fill_header
        ws.cell(row=2, column=col).font = bold_font
        ws.cell(row=2, column=col).fill = fill_header

    for merge_range in ws.merged_cells.ranges:
        for row_cells in ws[merge_range.coord]:
            for cell in row_cells:
                cell.alignment = center_align
                cell.font = bold_font
                cell.fill = fill_header

    for r in summary_rows:
        for col in range(1, ws.max_column + 1):
            cell = ws.cell(row=r, column=col)
            cell.font = bold_font
            cell.fill = fill_summary

    for r in range(1, ws.max_row + 1):
        for col in range(4, 7):  # D,E,F列
            cell = ws.cell(row=r, column=col)
            cell.fill = fill_weekly

    ws.freeze_panes = ws["G3"]
    for col_idx in range(1, ws.max_column + 1):
        col_letter = get_column_letter(col_idx)
        ws.column_dimensions[col_letter].width = 12

    for r in range(3, ws.max_row + 1):
        ws[f"F{r}"].number_format = '0.00%'
        for rate_col in rate_cols:
            ws[f"{rate_col}{r}"].number_format = '0.00%'

beautify(ws_cn, summary_rows_cn, rate_cols)
beautify(ws_en, summary_rows_en, rate_cols_en)

save_path = "日报周度模板_中英文双表头_数据同步_无综合目标_含汇总行.xlsx"
wb.save(save_path)
print("已保存到：", save_path)
